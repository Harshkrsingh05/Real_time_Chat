rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Users collection - for profile and search
    match /users/{userId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && request.auth.uid == userId;
      allow update: if isAuthenticated() && request.auth.uid == userId;
      allow delete: if false;
    }
    
    // General messages (public chat room)
    match /generalMessages/{messageId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && 
                   request.auth.uid == request.resource.data.senderId ||
                   request.resource.data.senderEmail == request.auth.token.email;
      allow update, delete: if false;
    }
    
    // Chat rooms (direct messages)
    match /chatRooms/{chatRoomId} {
      allow read: if isAuthenticated() && 
                 request.auth.uid in resource.data.participants;
      
      allow create: if isAuthenticated() && 
                   request.auth.uid in request.resource.data.participants;
      
      allow update: if isAuthenticated() && 
                   request.auth.uid in resource.data.participants;
      
      allow delete: if isAuthenticated() && 
                   request.auth.uid == resource.data.createdBy;
    }
    
    // Messages (direct chat messages)
    match /messages/{messageId} {
      // Allow read if authenticated (will filter by chatRoomId in query)
      allow read: if isAuthenticated();
      
      // Allow create if authenticated and sender matches
      allow create: if isAuthenticated() && 
                   request.auth.uid == request.resource.data.senderId;
      
      // Allow update/delete of own messages
      allow update, delete: if isAuthenticated() && 
                           request.auth.uid == resource.data.senderId;
    }
  }
}
